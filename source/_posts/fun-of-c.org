#include <stdlib.h>
#include <stdio.h>

typedef struct NODE {
    struct NODE *link;
    int value;
} Node;

int insert(Node **linkp, int new_value)
{
    Node *current;
    Node *new;

    while( ( current = *linkp ) != NULL 
            &&  current->value < new_value )
    {
        linkp = &current->link;
    }

    new = (Node *)malloc(sizeof(Node));
    if(NULL == new)
    {
        return -1;
    }

    new->value = new_value;

    new->link = current;
    *linkp = new;

    return 1;

}

void printLink(Node **root)
{
    while(*root != NULL)
    {
        printf("%d ", (*root)->value);
        *root = ((*root)->link);
    }
}
